init {
    a, b, c : Int
    x, y : Float
    msg, name : String
}

#+ ----- Asignaciones simples: Int, Float, String, constante negativa ----- +#
a := 10
b := -5
x := 3.5
msg := "hola mundo"
name := msg

#+ ----- Aritmetica: suma, resta, producto, division, div, mod ----- +#
c := a + b * 2
c := a - b
c := a * 3
c := a / b
c := a div 2
c := a mod 3

#+ ----- Promocion valida: Float := Int ----- +#
x := a

#+ ----- Seleccion: IF simple ----- +#
if (a > b) {
    c := a
}

#+ ----- Seleccion: IF - ELSE ----- +#
if (a == b) {
    c := 1
} else {
    c := 0
}

#+ ----- IF anidado: seleccion dentro de seleccion ----- +#
if (a > 0) {
    if (b < 0) {
        c := a - b
    }
}

#+ ----- Condicion compuesta con AND ----- +#
if (a > 0 AND b < 0) {
    c := a
}

#+ ----- Ciclo WHILE ----- +#
while (a > 0) {
    a := a - 1
}

#+ ----- Ciclo FOR ascendente (tema especial) ----- +#
for c := 0 to 5 step 1
    a := a + c
next c

#+ ----- Ciclo FOR descendente, step negativo (tema especial) ----- +#
for c := 10 to 0 step -1
    a := a + 1
next c

#+ ----- Entrada y salida ----- +#
read(a)
write("resultado:")
write(a)

   este comentario externo #+ contiene uno interno +# y luego continua +#
